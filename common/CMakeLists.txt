# Make a variable with include directories for common directories in PARENT_SCOPE
file(GLOB_RECURSE COMMON_DIRS LIST_DIRECTORIES true ${CMAKE_CURRENT_LIST_DIR}/*)
list(INSERT COMMON_DIRS 0 ${CMAKE_CURRENT_LIST_DIR})
set(COMMON_DIRS ${COMMON_DIRS} PARENT_SCOPE)

# Make a variable for the common sources in PARENT_SCOPE
file(GLOB_RECURSE COMMON_SOURCES CONFIGURE_DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp ${CMAKE_CURRENT_SOURCE_DIR}/*.h)
set(COMMON_SOURCES ${COMMON_SOURCES} PARENT_SCOPE)
